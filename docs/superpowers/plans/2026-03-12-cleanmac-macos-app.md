# CleanMac macOS Menu Bar App - Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a native macOS menu bar app that wraps the existing `clean_mac.command` script with a SwiftUI UI featuring task selection, real-time logs, and disk space reporting.

**Architecture:** SwiftUI `MenuBarExtra` with `.window` style, MVVM pattern. `CleanerService` (plain class) executes shell commands via `Process` off the main thread using `Task.detached`. Privilege escalation via `osascript`. Shared `Theme` enum for consistent colors.

**Tech Stack:** Swift 5.9+, SwiftUI, macOS 13+ `MenuBarExtra` API, Foundation `Process`/`Pipe`, XcodeGen

### Task 3: Create CleanerService

**Files:**
- Create: `CleanMac/CleanMac/Services/CleanerService.swift`

Key design decisions:
- **Plain `class`**, not `actor` — `Process.waitUntilExit()` is synchronous and would block an actor's serial executor. The ViewModel calls service methods from `Task.detached` instead.
- **User cache uses `FileManager` directly** — `~/Library/Caches/*` is user-owned, no sudo needed.
- **System cache/logs use `runPrivileged`** — requires elevated privileges via osascript.
- **Dev tools are conditional** — skip if brew/npm/pip not found, using `skipped` status.

- [ ] **Step 1: Write CleanerService**

Create `CleanMac/CleanMac/Services/CleanerService.swift`:

```swift
import Foundation

class CleanerService {
    private let homeDir: String

    init() {
        self.homeDir = NSHomeDirectory()
    }

    // MARK: - Disk Space

    func getAvailableDiskSpace() -> String {
        let result = runShell("/bin/df", arguments: ["-h", "/"])
        let lines = result.split(separator: "\n")
        guard lines.count >= 2 else { return "unknown" }
        let columns = lines[1].split(separator: " ", omittingEmptySubsequences: true)
        guard columns.count >= 4 else { return "unknown" }
        return String(columns[3])
    }

    // MARK: - Task 1: Time Machine Snapshots

    func runTimeMachineCleanup(onLog: @Sendable (String, Bool) -> Void) throws {
        onLog("正在清理 Time Machine 本地快照...", false)
        try runPrivileged("tmutil thinlocalsnapshots / 1000000000000 4")
        onLog("✓ 快照清理完成", false)
    }

    // MARK: - Task 2: System Cache and Logs

    func runCacheCleanup(onLog: @Sendable (String, Bool) -> Void) throws {
        // User cache — user-owned, no sudo needed
        onLog("正在清理用户缓存...", false)
        let userCachePath = "\(homeDir)/Library/Caches"
        if let contents = try? FileManager.default.contentsOfDirectory(atPath: userCachePath) {
            for item in contents {
                let fullPath = "\(userCachePath)/\(item)"
                try? FileManager.default.removeItem(atPath: fullPath)
            }
        }
        onLog("✓ 用户缓存已清理", false)

        // System cache + old logs — batch into single privileged call to avoid multiple password prompts
        onLog("正在清理系统缓存和旧日志（需要管理员权限）...", false)
        var cmds: [String] = []
        let safeDirs = [
            "/Library/Caches/com.apple.Safari",
            "/Library/Caches/com.apple.dt.Xcode",
            "/Library/Caches/Homebrew",
        ]
        for dir in safeDirs {
            cmds.append("rm -rf '\(dir)/'* 2>/dev/null || true")
        }
        cmds.append("find /private/var/log -name '*.log' -mtime +7 -delete 2>/dev/null || true")
        try runPrivileged(cmds.joined(separator: " ; "))
        onLog("✓ 系统缓存已清理（仅安全目录）", false)

        // User logs — no sudo needed
        let userLogsPath = "\(homeDir)/Library/Logs"
        _ = runShell("/usr/bin/find", arguments: [userLogsPath, "-type", "f", "-mtime", "+7", "-delete"])
        onLog("✓ 7 天前的旧日志已清理", false)
    }

    // MARK: - Task 3: Developer Tool Cache

    /// Returns `true` if any cleanup was performed, `false` if all tools were skipped.
    func runDevToolsCleanup(onLog: @Sendable (String, Bool) -> Void) throws -> Bool {
        var didSomething = false

        // Homebrew
        let brewPaths = ["/opt/homebrew/bin/brew", "/usr/local/bin/brew"]
        if let brewPath = brewPaths.first(where: { FileManager.default.isExecutableFile(atPath: $0) }) {
            onLog("正在清理 Homebrew 缓存...", false)
            let process = Process()
            process.executableURL = URL(fileURLWithPath: brewPath)
            process.arguments = ["cleanup", "--prune=all"]
            process.standardOutput = FileHandle.nullDevice
            process.standardError = FileHandle.nullDevice
            try? process.run()
            process.waitUntilExit()
            onLog("✓ Homebrew 缓存已清理", false)
            didSomething = true
        }

        // npm
        let npmCache = "\(homeDir)/.npm/_cacache"
        if FileManager.default.fileExists(atPath: npmCache) {
            onLog("正在清理 npm 缓存...", false)
            try? FileManager.default.removeItem(atPath: npmCache)
            onLog("✓ npm 缓存已清理", false)
            didSomething = true
        }

        // pip
        let pipCache = "\(homeDir)/Library/Caches/pip"
        if FileManager.default.fileExists(atPath: pipCache) {
            onLog("正在清理 pip 缓存...", false)
            try? FileManager.default.removeItem(atPath: pipCache)
            onLog("✓ pip 缓存已清理", false)
            didSomething = true
        }

        if !didSomething {
            onLog("未检测到开发工具缓存，已跳过", false)
        }
        return didSomething
    }

    // MARK: - Task 4: Large File Scan

    func runLargeFileScan(onLog: @Sendable (String, Bool) -> Void) {
        onLog("正在扫描大文件...", false)
        let downloadsPath = "\(homeDir)/Downloads"
        let result = runShell(
            "/usr/bin/find",
            arguments: [downloadsPath, "-type", "f", "-mtime", "+7", "-size", "+200M"]
        )
        let files = result.split(separator: "\n").map(String.init).filter { !$0.isEmpty }

        if files.isEmpty {
            onLog("未发现可清理的大文件", false)
        } else {
            onLog("找到 \(files.count) 个大文件（>200MB，7天前）：", true)
            for file in files {
                let url = URL(fileURLWithPath: file)
                let name = url.lastPathComponent
                if let attrs = try? FileManager.default.attributesOfItem(atPath: file),
                   let size = attrs[.size] as? UInt64
                {
                    let sizeStr = ByteCountFormatter.string(fromByteCount: Int64(size), countStyle: .file)
                    onLog("  \(sizeStr)\t\(name)", true)
                }
            }
        }
        onLog("提示: 请手动检查 ~/Downloads 文件夹", true)
    }

    // MARK: - Shell Helpers

    private func runShell(_ command: String, arguments: [String] = []) -> String {
        let process = Process()
        let pipe = Pipe()
        process.executableURL = URL(fileURLWithPath: command)
        process.arguments = arguments
        process.standardOutput = pipe
        process.standardError = pipe
        try? process.run()
        // Read pipe data BEFORE waitUntilExit to avoid deadlock if output exceeds pipe buffer
        let data = pipe.fileHandleForReading.readDataToEndOfFile()
        process.waitUntilExit()
        return String(data: data, encoding: .utf8)?.trimmingCharacters(in: .whitespacesAndNewlines) ?? ""
    }

    private func runPrivileged(_ command: String) throws {
        let escaped = command.replacingOccurrences(of: "\\", with: "\\\\")
            .replacingOccurrences(of: "\"", with: "\\\"")
        let script = "do shell script \"\(escaped)\" with administrator privileges"
        let process = Process()
        process.executableURL = URL(fileURLWithPath: "/usr/bin/osascript")
        process.arguments = ["-e", script]
        let errPipe = Pipe()
        process.standardError = errPipe
        try process.run()
        process.waitUntilExit()
        if process.terminationStatus != 0 {
            let errData = errPipe.fileHandleForReading.readDataToEndOfFile()
            let errMsg = String(data: errData, encoding: .utf8) ?? "Unknown error"
            if errMsg.contains("User canceled") || errMsg.contains("-128") {
                throw CleanerError.userCancelled
            }
            throw CleanerError.taskFailed(errMsg)
        }
    }
}

enum CleanerError: LocalizedError {
    case userCancelled
    case taskFailed(String)

    var errorDescription: String? {
        switch self {
        case .userCancelled:
            return "用户取消了授权"
        case .taskFailed(let msg):
            return "任务失败: \(msg)"
        }
    }
}
```

## Chunk 2: ViewModel + Views + App Integration

### Task 4: Create CleanerViewModel

**Files:**
- Create: `CleanMac/CleanMac/ViewModels/CleanerViewModel.swift`

Key design decisions:
- **`@MainActor`** class with `@Published` properties for UI binding.
- **`Task.detached`** to run blocking `Process` calls off the MainActor — service methods are synchronous and would block the main thread otherwise.
- **`isCleaning`** published property drives the menu bar icon swap between static leaf and spinning animation.
- All UI updates from background via `await MainActor.run { ... }`.

- [ ] **Step 1: Write CleanerViewModel**

Create `CleanMac/CleanMac/ViewModels/CleanerViewModel.swift`:

```swift
import Foundation
import SwiftUI

enum AppState {
    case idle
    case cleaning
    case completed
}

@MainActor
class CleanerViewModel: ObservableObject {
    @Published var appState: AppState = .idle
    @Published var tasks: [CleanTask] = CleanTask.allTasks
    @Published var spaceBefore: String = ""
    @Published var spaceAfter: String = ""
    @Published var currentTaskIndex: Int = 0
    @Published var isCleaning: Bool = false
    @Published var totalLogCount: Int = 0

    private let service = CleanerService()

    var selectedTasks: [CleanTask] {
        tasks.filter { $0.isSelected }
    }

    var totalSelected: Int {
        selectedTasks.count
    }

    var completedCount: Int {
        tasks.filter { $0.isSelected }.filter {
            if case .completed = $0.status { return true }
            if case .failed = $0.status { return true }
            if case .skipped = $0.status { return true }
            return false
        }.count
    }

    func startCleanup() {
        guard !selectedTasks.isEmpty else { return }
        appState = .cleaning
        isCleaning = true
        currentTaskIndex = 0

        // Reset task states
        for i in tasks.indices {
            if tasks[i].isSelected {
                tasks[i].status = .pending
                tasks[i].logLines = []
            }
        }

        Task.detached { [service, weak self] in
            guard let self else { return }

            let spaceBefore = service.getAvailableDiskSpace()
            await MainActor.run { self.spaceBefore = spaceBefore }

            let tasksCopy = await MainActor.run { self.tasks }
            for i in tasksCopy.indices where tasksCopy[i].isSelected {
                await MainActor.run {
                    self.currentTaskIndex = self.completedCount + 1
                    self.tasks[i].status = .inProgress
                }

                let logHandler: @Sendable (String, Bool) -> Void = { text, isWarning in
                    Task { @MainActor in
                        self.tasks[i].logLines.append(LogLine(text, isWarning: isWarning))
                        self.totalLogCount += 1
                    }
                }

                do {
                    switch tasksCopy[i].id {
                    case "timemachine":
                        try service.runTimeMachineCleanup(onLog: logHandler)
                    case "cache":
                        try service.runCacheCleanup(onLog: logHandler)
                    case "devtools":
                        let didWork = try service.runDevToolsCleanup(onLog: logHandler)
                        if !didWork {
                            await MainActor.run { self.tasks[i].status = .skipped("未检测到开发工具缓存") }
                            continue
                        }
                    case "largefiles":
                        service.runLargeFileScan(onLog: logHandler)
                    default:
                        break
                    }
                    await MainActor.run { self.tasks[i].status = .completed }
                } catch CleanerError.userCancelled {
                    await MainActor.run {
                        self.tasks[i].status = .failed("用户取消了授权")
                        self.isCleaning = false
                        self.resetToIdle()
                    }
                    return
                } catch {
                    await MainActor.run {
                        self.tasks[i].status = .failed(error.localizedDescription)
                        self.tasks[i].logLines.append(
                            LogLine("✗ \(error.localizedDescription)", isWarning: true)
                        )
                    }
                    // Continue with remaining tasks
                }
            }

            let spaceAfter = service.getAvailableDiskSpace()
            await MainActor.run {
                self.spaceAfter = spaceAfter
                self.isCleaning = false
                self.appState = .completed
            }
        }
    }

    func resetToIdle() {
        appState = .idle
        isCleaning = false
        for i in tasks.indices {
            tasks[i].status = .pending
            tasks[i].logLines = []
            tasks[i].isSelected = true
        }
        spaceBefore = ""
        spaceAfter = ""
        currentTaskIndex = 0
    }
}
```

```swift
struct LogItemView: View {

    let status: TaskStatus

    private var icon: String {
        switch status {
        case .completed:
            return "✓"
        case .inProgress:
            return "›"
        case .pending:
            return "·"
        case .failed:
            return "✗"

            return "–"

        if line.isWarning {
            return Theme.warning
        }
        switch status {
        case .completed:
            return Theme.primary
        case .inProgress:
            return Theme.inProgress
        case .pending:
            return .secondary
        case .failed:
            return Theme.warning
        case .skipped:
            return .secondary
        }
    }

    var body: some View {
        HStack(alignment: .top, spacing: 6) {
            Text(icon)
                .foregroundColor(iconColor)
                .font(.system(.caption, design: .monospaced))
                .frame(width: 12)
            Text(line.text)
                .font(.system(.caption, design: .monospaced))
                .foregroundColor(line.isWarning ? Theme.warning : .primary)
        }
    }
}
```

```swift
                            ForEach(task.logLines) { line in

                .clipShape(RoundedRectangle(cornerRadius: 8))

                    withAnimation {
                        proxy.scrollTo("bottom")
                    }
                }
            }
        }
    }
}
```

```swift
import SwiftUI

@main
struct CleanMacApp: App {

    init() {
        NSApplication.shared.setActivationPolicy(.accessory)
    }

    var body: some Scene {
        MenuBarExtra {
            MenuBarView(viewModel: viewModel)
        } label: {
            if viewModel.isCleaning {
                if #available(macOS 14, *) {
                    Image(systemName: "arrow.triangle.2.circlepath")
                        .symbolEffect(.rotate, isActive: viewModel.isCleaning)
                } else {

            }
        }

    }
}
```

The `#available(macOS 14, *)` guard is required because `.symbolEffect(.rotate)` is macOS 14+ only while our deployment target is macOS 13. On macOS 13, the spinning icon displays as a static system icon — an acceptable graceful degradation since the progress bar inside the panel still indicates activity.

- [ ] **Step 3: Create icon generation script**

Create `CleanMac/generate-icons.sh`:

```bash
#!/bin/bash
set -euo pipefail

# Requires: brew install librsvg
command -v rsvg-convert >/dev/null 2>&1 || {
    echo "Error: rsvg-convert not found. Install with: brew install librsvg"
    exit 1
}

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
APP_ICON_DIR="$SCRIPT_DIR/CleanMac/Assets.xcassets/AppIcon.appiconset"
MENUBAR_DIR="$SCRIPT_DIR/CleanMac/Assets.xcassets/MenuBarIcon.imageset"

echo "Generating app icon PNGs..."
for size in 16 32 64 128 256 512 1024; do
    rsvg-convert -w "$size" -h "$size" "$SCRIPT_DIR/icon-source.svg" \
        -o "$APP_ICON_DIR/icon_${size}.png"
    echo "  icon_${size}.png"
done

echo "Generating menu bar icon PNGs..."
rsvg-convert -w 18 -h 18 "$SCRIPT_DIR/menubar-icon-source.svg" \
    -o "$MENUBAR_DIR/menubar_icon.png"
rsvg-convert -w 36 -h 36 "$SCRIPT_DIR/menubar-icon-source.svg" \
    -o "$MENUBAR_DIR/menubar_icon@2x.png"
echo "  menubar_icon.png, menubar_icon@2x.png"

# Verify key output files are non-empty
for f in "$APP_ICON_DIR/icon_1024.png" "$MENUBAR_DIR/menubar_icon.png"; do
    [ -s "$f" ] || { echo "Error: $f is empty or missing"; exit 1; }
done

echo "Done."
```
